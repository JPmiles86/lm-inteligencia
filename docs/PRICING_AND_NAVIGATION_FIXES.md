```typescript
const handlePricingClick = (e: React.MouseEvent<HTMLAnchorElement>) => {
  if (location.pathname === '/') {
    e.preventDefault();
    const pricingSection = document.getElementById('pricing');
    if (pricingSection) {
      pricingSection.scrollIntoView({ behavior: 'smooth' });
    }
  }
};
```
